# Criar partição primaria de 40G no hd de 120G
fdisk /dev/vdb <<EOF
n
p
1
2048
+40G
t
8e
p
w
EOF
# Creação do partiçção fisica
pvcreate /dev/vdb1

# Creação do grupo
vgcreate vgnew /dev/vdb1

# Extensão do volume lógico em percentagem
lvcreate -l 50%VG -n lvnew1 vgnew

# Crear volume logico com tamanho predeterminado "10G"
lvcreate -L 10G -n lvnew2 vgnew

# Formatação
mkfs.ext3 /dev/mapper/vgnew-lvnew2
mkfs.ext3 /dev/vgnew/lvnew2

# Redimensionar volume logico adicionando valor predeterminado "+5G"
lvresize -L +5G /dev/vgnew/lvnew2

# Atribuir todo espaço disponivel
lvresize -l +100%FREE /dev/vgnew/lvnew2

# Estender partição
fdisk /dev/vdb
# Comandos: p, d, n, p, 1, [MESMO START], [Enter], t, 8e, w

# Recarregar
partprobe /dev/vdb

# Redimensionar partição fisica
pvresize  /dev/vdb1



⚠️ Importante: Por que é Seguro?
LVM armazena metadados no início da partição

Ao manter o mesmo Start Sector, não mexemos nos dados existentes

Só estendemos o final da partição para o espaço não alocado

Os dados do LVM permanecem intacto
